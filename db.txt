create database goaskme_game;
use goaskme_game;
create table game_user(account VARCHAR(30), wallet VARCHAR(20),points FLOAT);
create table game_login(uuid VARCHAR(20), account VARCHAR(30), myTimestamp INT(20), ref VARCHAR(20), sign VARCHAR(120));
create table game_withdraw(account VARCHAR(30), GAMT_amount FLOAT, request_time timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP, memo VARCHAR(20));
create table game_snake(account VARCHAR(30), score INT, last_game_time timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP);
create table game_2048(account VARCHAR(30), score INT, last_game_time timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP);
create table game_hitball(account VARCHAR(30), score INT, last_game_time timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP);
create table game_bonus_pool(game_id INT, game_name VARCHAR(20), cumulative_points INT, update_time timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP);
